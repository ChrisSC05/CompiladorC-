// Example usage of built-ins
class Main {
    int   num;
    char  letter;
    int[]    arrI;
    string[] arrS;
    double[] arrD;
    char[]   arrC;
    bool[]   arrB;
    int i;

    void main() {
        // ord → convierte char a int
        num = ord('A');
        print(num);               // 65

        // chr → convierte int a char
        letter = chr(66);
        print(letter);            // B

        // add → agrega elementos a distintos arreglos
        arrI = new int[0];
        arrS = new string[0];
        arrD = new double[0];
        arrC = new char[0];
        arrB = new bool[0];
        add(arrI, 10);
        add(arrS, "hello");
        add(arrD, 3.14);
        add(arrC, 'x');
        add(arrB, true);

        // len → tamaño del arreglo después de add
        print(len(arrI));         // 1
        print(len(arrS));         // 1
        print(len(arrD));         // 1
        print(len(arrC));         // 1
        print(len(arrB));         // 1

        if (true) print(1); else print(0);

        i = 0;
        while (i < 3) {
            print(i);
            i = i + 1;
        }

        for (i = 0; i < 3; i = i + 1) {
            print(i);
        }
        
        //print del contenido de los arreglos
        print(arrI[0]);           // 10
        print(arrS[0]);           // hello
        print(arrD[0]);           // 3.14
        print(arrC[0]);           // x
        print(arrB[0]);           // true

        // del → elimina el primer elemento
        del(arrI, 0);
        del(arrS, 0);
        del(arrD, 0);
        del(arrC, 0);
        del(arrB, 0);

        // len de nuevo para verificar que se eliminaron
        print(len(arrI));         // 0
        print(len(arrS));         // 0
        print(len(arrD));         // 0
        print(len(arrC));         // 0
        print(len(arrB));         // 0
    }
}
